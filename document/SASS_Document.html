Responsive Design

Breakpoints
min-width:1025px
max-width:1024px
min-width:768px
max-width:767px
Use @media query
// Define @mixin code snippets
@mixin media ( $width, $type:max-width ) {
    @media only screen and ( $type: $width ) {
        @content;
    }
}

// Use @mixin code snippets in css
.class { 
    color: red;

    @if $forpc == true { // for PC
        @include media ( $media-lg ) { // max-width: 768px 
            color: blue;
        }
    }
    @else { // for Mobile
            color: yellow; // for iPad min-width: 768px 
        @include media ( $media-lm ) { // for iPhone max-width: 767px 
            color: orange;
        }
    }

}
